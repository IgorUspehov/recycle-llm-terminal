import pandas as pd
import sys

def monitor_amperage(csv_path, window=20, threshold=0.15):
    df = pd.read_csv(csv_path, parse_dates=['timestamp'])
    df['rolling_avg'] = df['amperage'].rolling(window=window).mean()
    df['deviation'] = (df['amperage'] - df['rolling_avg']) / df['rolling_avg']
    flags = df[df['deviation'] > threshold]
    if not flags.empty:
        print(f"ALERT: {len(flags)} deviation events detected")
        print(flags[['timestamp','amperage','rolling_avg','deviation']].to_string())
    else:
        print("OK: no deviations above threshold")

if __name__ == '__main__':
    monitor_amperage(sys.argv[1])
